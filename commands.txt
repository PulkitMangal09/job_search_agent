To run it entirely: python -m pipelines.full_pipeline --sources jsearch   
Set api key for jsearch : export RAPIDAPI_KEY=your_key_here
For search: python -m agent.searcher --query "senior kubernetes engineer remote" --top 5
Drop table: docker exec -it $(docker ps -qf "name=postgres") psql -U postgres -d job_agent -c "DROP TABLE IF EXISTS jobs CASCADE; DROP TABLE IF EXISTS skill_trends CASCADE;"
